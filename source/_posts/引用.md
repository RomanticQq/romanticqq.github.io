### 引用的基本使用

```c++
#include<iostream>
using namespace std;
int main()
{
	int a = 10;
	int& b = a;
	cout << a << endl;
	cout << b << endl;
	a = 20;
	cout << a << endl;//20
	cout << b << endl;//20
	b = 30;
	cout << a << endl;//30
	cout << b << endl;//30
	return 0;
}
```

> int& b = a就相当于给变量a起一个别名，a和b的值是一直保持相等的，只要一个改变，两个都改变；

### 引用的注意事项

```c++
#include<iostream>
using namespace std;
int main()
{
	int a = 10;
	int b = 20;
	//int& b;//这样是不对的  引用被定义时必须初始化
	int& c = a;
	c = b; //引用一旦初始化就不能被更改，这里是赋值操作
	cout << a << endl;//20
	cout << b << endl;//20
	cout << c << endl;//20
}
```

> - 引用被定义时必须初始化
> - 引用一旦被初始化就不能更改

### 引用做函数参数

```c++
#include<iostream>
using namespace std;
void mySwap01(int a, int b)
{
	int temp = a;
	a = b;
	b = temp;
}
void mySwap02(int* a, int* b)
{
	int temp = *a;
	*a = *b;
	*b = temp;
}
void mySwap03(int &a, int &b)
{
	int temp = a;
	a = b;
	b = temp;
}
int main()
{
	int a = 10;
	int b = 20;
	//mySwap01(a, b);//10,20
	//mySwap02(&a, &b);//20,10
	mySwap03(a, b);//20,10
	cout << a << endl;
	cout << b << endl;
}
```

> 引用传参和地址传参的结果是相同的。

### 引用做函数返回值

```c++
#include<iostream>
using namespace std;
//引用做函数的返回值
//1.不要返回局部变量的引用
int& test01() {
	int a = 10; //局部变量 存在在内存四区中的栈区
	return a;
}

//返回静态变量引用
int& test02() {
	static int a = 20;
	return a;
}

int main() {

	//不能返回局部变量的引用
	int& ref = test01();
	cout << "ref = " << ref << endl;//内存此时已经被释放，所以会出现乱码
	cout << "ref = " << ref << endl;//内存此时已经被释放，所以会出现乱码

	//如果函数做左值，那么必须返回引用
	int& ref2 = test02(); //当执行完这一行后，此时的test02()已经相当于是一个引用变量，类型是int&
	cout << "ref2 = " << ref2 << endl;
	cout << "ref2 = " << ref2 << endl;

	test02() = 1000;//如果函数的返回值是引用，那么这个函数调用可以作为左值

	cout << "ref2 = " << ref2 << endl;
	cout << "ref2 = " << ref2 << endl;
	cout << "test02 = " <<test02() << endl;

	system("pause");

	return 0;
}
```

>- 不能返回局部变量的引用；
>- 当函数做左值的时候，必须返回引用；

### 引用的本质

```c++
//发现是引用，转换为 int* const ref = &a;
void func(int& ref){
	ref = 100; // ref是引用，转换为*ref = 100
}
int main(){
	int a = 10;
    
    //自动转换为 int* const ref = &a; 指针常量是指针指向不可改，也说明为什么引用不可更改
	int& ref = a; 
	ref = 20; //内部发现ref是引用，自动帮我们转换为: *ref = 20;
    
	cout << "a:" << a << endl;
	cout << "ref:" << ref << endl;
    
	func(a);
	return 0;
}
```

> 引用的本质在c++内部实现是一个指针常量。

### 常量引用

```c++
//引用使用的场景，通常用来修饰形参
void showValue(const int& v) {
	//v += 10;
	cout << v << endl;
}

int main() {

	//int& ref = 10;  引用本身需要一个合法的内存空间，因此这行错误
	//加入const就可以了，编译器优化代码，int temp = 10; const int& ref = temp;
	const int& ref = 10;

	//ref = 100;  //加入const后不可以修改变量
	cout << ref << endl;

	//函数中利用常量引用防止误操作修改实参
	int a = 10;
	showValue(a);

	system("pause");

	return 0;
}
```

> 常量引用不可修改变量。
